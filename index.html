async function updateSignageData() { try { // 外部サイトではなく、自リポジトリ内の data.json を読み込む（CORSを完全回避） // キャッシュによる停滞を防ぐため、末尾にタイムスタンプを付与 const response = await fetch('./data.json?t=' + new Date().getTime()); if (!response.ok) { throw new Error('JSONファイルの読み込みに失敗しました'); } const data = await response.json(); if (data && data.length > 0) { const current = data[0]; // 各項目の数値を反映 document.getElementById('val-wbgt').textContent = Number(current.Wbgt).toFixed(1); document.getElementById('val-kion').textContent = Number(current.Kion).toFixed(1); document.getElementById('val-koku').textContent = Number(current.Koku).toFixed(1); document.getElementById('val-shitu').textContent = Number(current.Shitu).toFixed(1); // 各項目の「日付現時刻」を反映 document.getElementById('time-wbgt').textContent = current.Time; document.getElementById('time-kion').textContent = current.Time; document.getElementById('time-koku').textContent = current.Time; document.getElementById('time-shitu').textContent = current.Time; // WBGT背景色アラート const wbgtBlock = document.getElementById('block-wbgt'); if (current.Wbgt >= 31.0) { wbgtBlock.style.backgroundColor = '#800000'; } else if (current.Wbgt >= 28.0) { wbgtBlock.style.backgroundColor = '#664400'; } else { wbgtBlock.style.backgroundColor = '#000000'; } } } catch (error) { console.error('データ取得エラー:', error); } } // 初回実行と定期チェック updateSignageData(); // サイネージ側は30秒〜1分おきに data.json の更新をチェックしに行けば十分です setInterval(updateSignageData, 30000);
